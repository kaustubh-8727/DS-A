#include<bits/stdc++.h>
using namespace std;

int main(){
    int i,j,n,k;
    cin>>n;
    vector<int> v(n);
    for(i=0;i<n;i++){
        cin>>v[i];
    }
    if(n==0){
        cout<<n;
        return 0;
    }
    i=0;
    for (j = 1; j < v.size(); j++) {
        if (v[j] != v[i]) {
            i++;
            v[i] = v[j];
        }
    }
    k=i+1;
    for(i=0;i<k;i++){
        cout<<v[i]<<" ";
    }
    return 0;
}
